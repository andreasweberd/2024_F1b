bool isPlaying = false;

void setup() {
  pinMode(buttonPin, INPUT_PULLUP);
  pinMode(buzzerPin, OUTPUT);
}

void loop() {
  int buttonState = digitalRead(buttonPin);

  if (buttonState == HIGH) {
    isPlaying = true;
  }

  if (isPlaying) {
    playMelody();
    isPlaying = false;
  }
}

void playMelody() {
  int melodyLength = sizeof(melody) / sizeof(melody[0]);

  for (int i = 0; i < melodyLength; i++) {
    int note = melody[i][0];
    int duration = melody[i][1];

    if (note == NOTE_REST) {
      noTone(buzzerPin);
    } else {
      tone(buzzerPin, note, duration);
    }

    delay(duration * 1.1);  // Verlangsamung der Note fÃ¼r besseren Klang
    noTone(buzzerPin);
  }
}
